#include <stdio.h>
#include <time.h>

void printSubset(int subset[], int size) {

    printf("{ ");

    for (int i = 0; i < size; i++) {
        printf("%d ", subset[i]);
    }

    printf("}\n");
}

void findSubsets(int arr[], int n,
                 int index, int sum,
                 int subset[], int size) {

    if (sum == 0) {

        printSubset(subset, size);
        return;
    }

    for (int i = index; i < n; i++) {

        if (arr[i] <= sum) {

            subset[size] = arr[i];

            findSubsets(arr, n,
                        i + 1,
                        sum - arr[i],
                        subset,
                        size + 1);
        }
    }
}

int main() {

    int n, i, sum;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter elements:\n");

    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter target sum: ");
    scanf("%d", &sum);

    int subset[n];

    printf("Possible Subsets:\n");

    clock_t start, end;

    start = clock();

    findSubsets(arr, n, 0, sum, subset, 0);

    end = clock();

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
