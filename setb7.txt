Writer File: setb7_writer.c
#include <stdio.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>

int main() {

    key_t key = 1234;

    int shmid = shmget(key, sizeof(int), 0666 | IPC_CREAT);

    int *counter = (int*) shmat(shmid, NULL, 0);

    *counter = 0;

    for(int i=0; i<10; i++) {

        (*counter)++;

        printf("Writer: %d\n", *counter);

        sleep(1);
    }

    shmdt(counter);

    return 0;
}
Reader File: setb7_reader.c
#include <stdio.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>

int main() {

    key_t key = 1234;

    int shmid = shmget(key, sizeof(int), 0666);

    int *counter = (int*) shmat(shmid, NULL, 0);

    for(int i=0; i<10; i++) {

        printf("Reader: %d\n", *counter);

        sleep(1);
    }

    shmdt(counter);

    shmctl(shmid, IPC_RMID, NULL);

    return 0;
}
