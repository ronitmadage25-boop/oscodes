#include <stdio.h>

struct Segment {

    int base;
    int limit;
};

int main() {

    struct Segment seg[3] = {
        {1000,400},
        {2000,500},
        {3000,300}
    };

    int s[] = {0,1,2};

    int offset[] = {200,600,150};

    for(int i=0; i<3; i++) {

        if(offset[i] < seg[s[i]].limit) {

            int physical = seg[s[i]].base + offset[i];

            printf("Physical Address = %d\n", physical);
        }

        else {

            printf("Segmentation fault for Segment %d\n", s[i]);
        }
    }

    return 0;
}
