#include <stdio.h>

int main() {

    int bt[5] = {6,4,2,3,1};
    int at[5] = {0,1,2,3,4};

    int wt[5], tat[5];

    wt[0] = 0;

    for(int i=1; i<5; i++)
        wt[i] = wt[i-1] + bt[i-1];

    for(int i=0; i<5; i++)
        tat[i] = wt[i] + bt[i];

    float avg = 0;

    printf("Process\tWT\tTAT\n");

    for(int i=0; i<5; i++) {

        printf("P%d\t%d\t%d\n", i+1, wt[i], tat[i]);

        avg += wt[i];
    }

    avg /= 5;

    printf("\nAverage Waiting Time = %.2f\n", avg);

    printf("\nGantt Chart:\n");

    for(int i=0; i<5; i++)
        printf("|---P%d---", i+1);

    printf("|\n");

    return 0;
}
