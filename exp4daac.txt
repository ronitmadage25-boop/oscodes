#include <stdio.h>
#include <time.h>

int max(int a, int b) {
    if (a > b)
        return a;
    return b;
}

int main() {

    int n, W, i, w;

    printf("Enter number of items: ");
    scanf("%d", &n);

    int wt[n], val[n];

    printf("Enter weights:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &wt[i]);
    }

    printf("Enter values:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &val[i]);
    }

    printf("Enter capacity: ");
    scanf("%d", &W);

    int dp[n + 1][W + 1];

    for (i = 0; i <= n; i++) {
        for (w = 0; w <= W; w++) {
            dp[i][w] = 0;
        }
    }

    clock_t start, end;

    start = clock();

    for (i = 1; i <= n; i++) {

        for (w = 1; w <= W; w++) {

            if (wt[i - 1] <= w) {

                dp[i][w] = max(
                    val[i - 1] + dp[i - 1][w - wt[i - 1]],
                    dp[i - 1][w]
                );

            } else {

                dp[i][w] = dp[i - 1][w];
            }
        }
    }

    end = clock();

    printf("Maximum Profit: %d\n", dp[n][W]);

    double runningTime = ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
