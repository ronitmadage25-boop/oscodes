#include <stdio.h>

struct Process {

    int pid, bt, pr, wt, tat;
};

int main() {

    struct Process p[5] = {
        {1,4,2},
        {2,3,1},
        {3,5,3},
        {4,1,1},
        {5,2,4}
    };

    for(int i=0; i<5; i++) {

        for(int j=i+1; j<5; j++) {

            if(p[i].pr > p[j].pr) {

                struct Process temp = p[i];
                p[i] = p[j];
                p[j] = temp;
            }
        }
    }

    p[0].wt = 0;

    for(int i=1; i<5; i++)
        p[i].wt = p[i-1].wt + p[i-1].bt;

    float avgwt = 0, avgtat = 0;

    printf("PID\tBT\tPR\tWT\tTAT\n");

    for(int i=0; i<5; i++) {

        p[i].tat = p[i].wt + p[i].bt;

        avgwt += p[i].wt;
        avgtat += p[i].tat;

        printf("P%d\t%d\t%d\t%d\t%d\n",
               p[i].pid,
               p[i].bt,
               p[i].pr,
               p[i].wt,
               p[i].tat);
    }

    printf("\nAverage WT = %.2f\n", avgwt/5);

    printf("Average TAT = %.2f\n", avgtat/5);

    return 0;
}
