#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <ctype.h>

int main() {

    int fd[2];

    char msg[] = "Hello from parent";

    char buffer[100];

    pipe(fd);

    if(fork() == 0) {

        close(fd[1]);

        read(fd[0], buffer, sizeof(buffer));

        for(int i=0; buffer[i]; i++)
            buffer[i] = toupper(buffer[i]);

        printf("Child received: %s\n", buffer);

        close(fd[0]);
    }

    else {

        close(fd[0]);

        write(fd[1], msg, strlen(msg)+1);

        close(fd[1]);
    }

    return 0;
}
