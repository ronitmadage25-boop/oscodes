Writer File: setc7_writer.c
#include <stdio.h>
#include <fcntl.h>
#include <sys/mman.h>
#include <unistd.h>

int main() {

    int fd = shm_open("/score", O_CREAT | O_RDWR, 0666);

    ftruncate(fd, sizeof(int));

    int *score = mmap(0, sizeof(int),
                      PROT_WRITE,
                      MAP_SHARED,
                      fd, 0);

    *score = 100;
    printf("Writer: %d\n", *score);
    sleep(1);

    *score = 200;
    printf("Writer: %d\n", *score);
    sleep(1);

    *score = 300;
    printf("Writer: %d\n", *score);

    munmap(score, sizeof(int));

    return 0;
}
Reader File: setc7_reader.c
#include <stdio.h>
#include <fcntl.h>
#include <sys/mman.h>
#include <unistd.h>

int main() {

    int fd = shm_open("/score", O_RDONLY, 0666);

    int *score = mmap(0, sizeof(int),
                      PROT_READ,
                      MAP_SHARED,
                      fd, 0);

    for(int i=0; i<3; i++) {

        printf("Reader: %d\n", *score);

        sleep(1);
    }

    munmap(score, sizeof(int));

    shm_unlink("/score");

    return 0;
}
