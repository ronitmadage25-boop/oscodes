#include <stdio.h>
#include <stdbool.h>
#include <time.h>

int board[8][8];

bool isSafe(int row, int col) {

    int i, j;

    for (i = 0; i < col; i++) {

        if (board[row][i] == 1) {
            return false;
        }
    }

    for (i = row, j = col;
         i >= 0 && j >= 0;
         i--, j--) {

        if (board[i][j] == 1) {
            return false;
        }
    }

    for (i = row, j = col;
         i < 8 && j >= 0;
         i++, j--) {

        if (board[i][j] == 1) {
            return false;
        }
    }

    return true;
}

bool solve(int col) {

    if (col >= 8) {
        return true;
    }

    for (int i = 0; i < 8; i++) {

        if (isSafe(i, col)) {

            board[i][col] = 1;

            if (solve(col + 1)) {
                return true;
            }

            board[i][col] = 0;
        }
    }

    return false;
}

int main() {

    clock_t start, end;

    start = clock();

    bool result = solve(0);

    end = clock();

    if (result) {

        printf("Solution Matrix:\n");

        for (int i = 0; i < 8; i++) {

            for (int j = 0; j < 8; j++) {

                printf("%d ", board[i][j]);
            }

            printf("\n");
        }

    } else {

        printf("No Solution Found\n");
    }

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
