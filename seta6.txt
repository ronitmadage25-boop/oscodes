#include <stdio.h>

int main() {

    int alloc[4][2] = {
        {1,0},
        {2,1},
        {1,1},
        {0,2}
    };

    int max[4][2] = {
        {3,2},
        {2,2},
        {3,3},
        {1,2}
    };

    int avail[2] = {2,1};

    int need[4][2];

    int finish[4] = {0};

    int safe[4];

    for(int i=0; i<4; i++)
        for(int j=0; j<2; j++)
            need[i][j] = max[i][j] - alloc[i][j];

    int count = 0;

    while(count < 4) {

        int found = 0;

        for(int i=0; i<4; i++) {

            if(finish[i] == 0) {

                if(need[i][0] <= avail[0] &&
                   need[i][1] <= avail[1]) {

                    avail[0] += alloc[i][0];
                    avail[1] += alloc[i][1];

                    safe[count++] = i;

                    finish[i] = 1;

                    found = 1;
                }
            }
        }

        if(found == 0)
            break;
    }

    if(count == 4) {

        printf("Safe sequence: ");

        for(int i=0; i<4; i++)
            printf("P%d ", safe[i]);
    }
    else {
        printf("System is in deadlock");
    }

    return 0;
}
