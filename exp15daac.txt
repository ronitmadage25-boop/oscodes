#include <stdio.h>
#include <time.h>

int main() {

    int puzzle[4][4];

    printf("Enter 4x4 Puzzle Matrix:\n");

    for (int i = 0; i < 4; i++) {

        for (int j = 0; j < 4; j++) {

            scanf("%d", &puzzle[i][j]);
        }
    }

    clock_t start, end;

    start = clock();

    printf("Puzzle Matrix:\n");

    for (int i = 0; i < 4; i++) {

        for (int j = 0; j < 4; j++) {

            printf("%d ", puzzle[i][j]);
        }

        printf("\n");
    }

    printf("15 Puzzle uses Branch and Bound technique.\n");

    end = clock();

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
