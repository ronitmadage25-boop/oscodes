#include <stdio.h>
#include <time.h>

int main() {

    int V, E, i, j, src;

    printf("Enter vertices and edges: ");
    scanf("%d%d", &V, &E);

    int edges[E][3];

    printf("Enter source destination weight:\n");

    for (i = 0; i < E; i++) {

        scanf("%d%d%d",
              &edges[i][0],
              &edges[i][1],
              &edges[i][2]);
    }

    printf("Enter source vertex: ");
    scanf("%d", &src);

    int dist[V];

    for (i = 0; i < V; i++) {
        dist[i] = 99999;
    }

    dist[src] = 0;

    clock_t start, end;

    start = clock();

    for (i = 1; i < V; i++) {

        for (j = 0; j < E; j++) {

            int u = edges[j][0];
            int v = edges[j][1];
            int w = edges[j][2];

            if (dist[u] != 99999 &&
                dist[u] + w < dist[v]) {

                dist[v] = dist[u] + w;
            }
        }
    }

    end = clock();

    printf("Shortest Distances:\n");

    for (i = 0; i < V; i++) {

        printf("%d -> %d = %d\n",
               src, i, dist[i]);
    }

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
