#include <stdio.h>
#include <pthread.h>
#include <unistd.h>

pthread_mutex_t fork1[3];

void* philosopher(void* arg) {

    int id = *(int*)arg;

    int left = id;
    int right = (id + 1) % 3;

    printf("Philosopher %d is thinking\n", id);

    sleep(1);

    if(id == 0) {

        pthread_mutex_lock(&fork1[right]);
        pthread_mutex_lock(&fork1[left]);
    }

    else {

        pthread_mutex_lock(&fork1[left]);
        pthread_mutex_lock(&fork1[right]);
    }

    printf("Philosopher %d is eating\n", id);

    sleep(1);

    pthread_mutex_unlock(&fork1[left]);
    pthread_mutex_unlock(&fork1[right]);

    return NULL;
}

int main() {

    pthread_t p[3];

    int id[3] = {0,1,2};

    for(int i=0; i<3; i++)
        pthread_mutex_init(&fork1[i], NULL);

    for(int i=0; i<3; i++)
        pthread_create(&p[i], NULL, philosopher, &id[i]);

    for(int i=0; i<3; i++)
        pthread_join(p[i], NULL);

    return 0;
}
