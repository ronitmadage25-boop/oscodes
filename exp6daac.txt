#include <stdio.h>
#include <stdlib.h>
#include <time.h>

struct Node {
    char ch;
    int freq;
    struct Node *left, *right;
};

struct Node* createNode(char ch, int freq) {

    struct Node* newNode =
        (struct Node*) malloc(sizeof(struct Node));

    newNode->ch = ch;
    newNode->freq = freq;
    newNode->left = NULL;
    newNode->right = NULL;

    return newNode;
}

void sort(struct Node* arr[], int n) {

    int i, j;
    struct Node* temp;

    for (i = 0; i < n - 1; i++) {

        for (j = 0; j < n - i - 1; j++) {

            if (arr[j]->freq > arr[j + 1]->freq) {

                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void printCodes(struct Node* root, int code[], int top) {

    if (root->left == NULL && root->right == NULL) {

        printf("%c : ", root->ch);

        int i;
        for (i = 0; i < top; i++) {
            printf("%d", code[i]);
        }

        printf("\n");
        return;
    }

    if (root->left) {
        code[top] = 0;
        printCodes(root->left, code, top + 1);
    }

    if (root->right) {
        code[top] = 1;
        printCodes(root->right, code, top + 1);
    }
}

int main() {

    int n, i;

    printf("Enter number of characters: ");
    scanf("%d", &n);

    struct Node* arr[100];

    for (i = 0; i < n; i++) {

        char ch;
        int freq;

        printf("Enter character: ");
        scanf(" %c", &ch);

        printf("Enter frequency: ");
        scanf("%d", &freq);

        arr[i] = createNode(ch, freq);
    }

    clock_t start, end;

    start = clock();

    int size = n;

    while (size > 1) {

        sort(arr, size);

        struct Node* left = arr[0];
        struct Node* right = arr[1];

        struct Node* newNode =
            createNode('-', left->freq + right->freq);

        newNode->left = left;
        newNode->right = right;

        for (i = 2; i < size; i++) {
            arr[i - 2] = arr[i];
        }

        arr[size - 2] = newNode;

        size--;
    }

    struct Node* root = arr[0];

    end = clock();

    int code[100];

    printf("Huffman Codes:\n");

    printCodes(root, code, 0);

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
