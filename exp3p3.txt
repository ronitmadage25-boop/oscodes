#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/types.h>
int main()
{
 int x=100;
 printf("Value of x before fork: %d\n",x);
 int id=fork();
 if(id==0)
 {
 printf("Child process sees x as: %d\n",x);
 x=250;
 printf("Child changed x to: %d\n",x);
 }
 else
 {
 printf("Parent process sees x as: %d\n",x);
 x=500;
 printf("Parent changed x to: %d\n",x);
 }
 return 0;
}
