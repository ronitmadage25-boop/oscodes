#include <stdio.h>
#include <stdlib.h>

int main() {

    int req[] = {55,58,39,18,90,160,150,38,184};

    int n = 9;

    int head = 100;

    int total = 0;

    printf("Order: %d ", head);

    for(int i=0; i<n; i++) {

        for(int j=i+1; j<n; j++) {

            if(req[i] > req[j]) {

                int temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }
        }
    }

    for(int i=4; i>=0; i--) {

        total += abs(head - req[i]);

        head = req[i];

        printf("-> %d ", head);
    }

    total += abs(head - 0);

    head = 199;

    printf("-> Jump -> %d ", head);

    for(int i=n-1; i>=5; i--) {

        total += abs(head - req[i]);

        head = req[i];

        printf("-> %d ", head);
    }

    printf("\nTotal Head Movement = %d\n", total);

    return 0;
}
