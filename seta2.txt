complex.h

#ifndef COMPLEX_H
#define COMPLEX_H

typedef struct {
    float real;
    float imag;
} Complex;

Complex add_complex(Complex a, Complex b);
Complex mul_complex(Complex a, Complex b);

#endif


complex.c//seta2_complex.c

#include "complex.h"

Complex add_complex(Complex a, Complex b) {
    Complex result;
    result.real = a.real + b.real;
    result.imag = a.imag + b.imag;
    return result;
}

Complex mul_complex(Complex a, Complex b) {
    Complex result;

    result.real = (a.real * b.real) - (a.imag * b.imag);
    result.imag = (a.real * b.imag) + (a.imag * b.real);

    return result;
}

test.c//seta2_test.c

#include <stdio.h>
#include "complex.h"

int main() {

    Complex c1 = {2, 3};
    Complex c2 = {4, 5};

    Complex sum = add_complex(c1, c2);
    Complex mul = mul_complex(c1, c2);

    printf("Addition = %.1f + %.1fi\n", sum.real, sum.imag);

    printf("Multiplication = %.1f + %.1fi\n", mul.real, mul.imag);

    return 0;
}


Commands

Compile object file:

gcc -c seta2_complex.c -o complex.o

Create static library:

ar rcs libcomplex.a complex.o

Compile test program:

gcc seta2_test.c -L. -lcomplex -o test

Run:

./test
