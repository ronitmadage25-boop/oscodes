#include <stdio.h>
#include <stdlib.h>

int main() {

    int req[] = {176,79,34,60,92,11,41,114};

    int n = 8;

    int head = 50;

    int total = 0;

    printf("SCAN Order: %d ", head);

    for(int i=0; i<n; i++) {

        for(int j=i+1; j<n; j++) {

            if(req[i] > req[j]) {

                int temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }
        }
    }

    int pos;

    for(int i=0; i<n; i++) {

        if(req[i] > head) {

            pos = i;
            break;
        }
    }

    for(int i=pos; i<n; i++) {

        total += abs(head - req[i]);

        head = req[i];

        printf("-> %d ", head);
    }

    for(int i=pos-1; i>=0; i--) {

        total += abs(head - req[i]);

        head = req[i];

        printf("-> %d ", head);
    }

    printf("\nTotal Head Movement = %d\n", total);

    return 0;
}
