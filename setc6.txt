#include <stdio.h>

int main() {

    int alloc[4][3] = {
        {0,1,0},
        {2,0,0},
        {3,0,2},
        {2,1,1}
    };

    int max[4][3] = {
        {7,5,3},
        {3,2,2},
        {9,0,2},
        {2,2,2}
    };

    int avail[3] = {3,3,2};

    int request[3] = {1,0,2};

    int need[4][3];

    for(int i=0; i<4; i++)
        for(int j=0; j<3; j++)
            need[i][j] = max[i][j] - alloc[i][j];

    int process = 1;

    int valid = 1;

    for(int i=0; i<3; i++) {

        if(request[i] > need[process][i] ||
           request[i] > avail[i]) {

            valid = 0;
        }
    }

    if(valid)
        printf("Request granted\n");

    else
        printf("Request denied\n");

    return 0;
}
