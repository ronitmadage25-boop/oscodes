#include <stdio.h>
#include <time.h>

struct Item {
    int weight;
    int value;
    float ratio;
};

void sort(struct Item items[], int n) {
    int i, j;
    struct Item temp;

    for (i = 0; i < n - 1; i++) {

        for (j = 0; j < n - i - 1; j++) {

            if (items[j].ratio < items[j + 1].ratio) {

                temp = items[j];
                items[j] = items[j + 1];
                items[j + 1] = temp;
            }
        }
    }
}

int main() {

    int n, i, cap;

    printf("Enter number of items: ");
    scanf("%d", &n);

    struct Item items[n];

    for (i = 0; i < n; i++) {

        printf("Enter weight and value: ");
        scanf("%d%d", &items[i].weight, &items[i].value);

        items[i].ratio =
            (float) items[i].value / items[i].weight;
    }

    printf("Enter capacity: ");
    scanf("%d", &cap);

    clock_t start, end;

    start = clock();

    sort(items, n);

    float profit = 0;

    for (i = 0; i < n; i++) {

        if (cap >= items[i].weight) {

            cap -= items[i].weight;
            profit += items[i].value;

        } else {

            profit += items[i].ratio * cap;
            break;
        }
    }

    end = clock();

    printf("Maximum Profit: %.2f\n", profit);

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
