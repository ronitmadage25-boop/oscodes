#include <stdio.h>
#include <limits.h>
#include <stdbool.h>
#include <time.h>

int main() {

    int n, i, j, src;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    int graph[n][n];

    printf("Enter adjacency matrix:\n");

    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &graph[i][j]);
        }
    }

    printf("Enter source vertex: ");
    scanf("%d", &src);

    int dist[n];
    bool visited[n];

    for (i = 0; i < n; i++) {
        dist[i] = INT_MAX;
        visited[i] = false;
    }

    dist[src] = 0;

    clock_t start, end;

    start = clock();

    for (int count = 0; count < n - 1; count++) {

        int u = -1;
        int min = INT_MAX;

        for (i = 0; i < n; i++) {

            if (!visited[i] && dist[i] < min) {
                min = dist[i];
                u = i;
            }
        }

        visited[u] = true;

        for (int v = 0; v < n; v++) {

            if (!visited[v] &&
                graph[u][v] != 0 &&
                dist[u] != INT_MAX &&
                dist[u] + graph[u][v] < dist[v]) {

                dist[v] = dist[u] + graph[u][v];
            }
        }
    }

    end = clock();

    printf("Shortest Distances:\n");

    for (i = 0; i < n; i++) {
        printf("%d -> %d = %d\n", src, i, dist[i]);
    }

    double runningTime =
        ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("Running Time: %f seconds\n", runningTime);

    return 0;
}
